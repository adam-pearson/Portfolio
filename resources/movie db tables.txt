Return all movies, with the actors, of all films with a higher average rating than American Beauty

SELECT *
FROM movie as mv
JOIN movie_direction AS mv_dr ON mv_dr.mov_id = mv.mov_id
JOIN director AS dr ON dr.dir_id = mv_dr.dir_id
